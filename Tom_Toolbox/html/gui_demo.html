function pushbutton8_Callback(hObject, eventdata, handles)
% hObject    handle to pushbutton8 (see GCBO)
% eventdata  reserved - to be defined in a future version of MATLAB
% handles    structure with handles and user data (see GUIDATA)
data=guidata(hObject);
boxsize = data.edit26;
pxsz = data.edit27;
rad = data.edit28;
height = data.edit29;
name = data.edit30;
box = str2double(boxsize.String);
listbox2 = data.listbox2;
if listbox2.Value == 1
    sphere = tom_spheremask(ones(box,box,box), str2double(rad.String),1,[round(box/2),round(box/2),round(box/2)]);
    tom_mrcwrite(sphere, 'name', name.String)
elseif listbox2.Value == 2
    curve = 9649;
    offset = 0;
    cyl=tom_cylindermask(ones([box,box,box]),str2double(rad.String),1,[round(box/2),round(box/2)]);
    sph_top=(tom_spheremask(ones([box,box,box]),curve,1,[round(box/2),round(box/2),round(box/2)+offset-round(str2double(height)/2)-curve])-1)*-1;
    sph_bot=(tom_spheremask(ones([box,box,box]),curve,1,[round(box/2),round(box/2),round(box/2)+offset+round(str2double(height)/2)+curve])-1)*-1;
    mask_final=cyl.*sph_top.*sph_bot;
    test = mask_geometric(box,0,str2double(rad.String),str2double(height.String),9649);
    %cyl = tom_cylindermask(, str2double(rad.String),1,[round(box/2),round(box/2)]);
    tom_mrcwrite(test, 'name', name.String)
end
if exist('chimera') == 1
    system(['alterheader testd4.mrc -del ',strcat(pxsz.String,',',pxsz.String,',',pxsz.String)])
elseif exist('chimera') == 0
    disp('Cannot find Chimera. Attempting to source sbgrid')
    system(['source /programs/sbgrid.shrc && alterheader testd4.mrc -del ',strcat(pxsz.String,',',pxsz.String,',',pxsz.String)])
end
